import { Meta } from "@storybook/addon-docs/blocks";
import React from 'react';

import NowPageView from "../components/NowPageView";
import { getLinearNowPageData } from "../data/linearNowPage";
import "../styles/now-page.css";

<Meta title="Astro Release Staging" />

export const Loader = () => {
  const [data, setData] = React.useState(null);
  const [error, setError] = React.useState(null);

  React.useEffect(() => {
    getLinearNowPageData()
      .then(setData)
      .catch(err => setError(err.message));
  }, []);

  if (error) {
    return (
      <div className="now-staging-error">
        <strong>Error loading live Linear data:</strong> {error}
      </div>
    );
  }

  if (!data) {
    return <div className="now-staging-loading">Analyzing Linear artifacts...</div>;
  }

  return (
    <div className="now-staging-shell">
      <div className="now-staging-intro">
        <p className="now-staging-kicker">Astro Release Staging // Release Candidate</p>
        <h1 className="now-staging-title">Dossier: /now</h1>
        <p className="now-staging-copy">
          This staging ground transposes the live <strong>www.tidelands.dev</strong> custom view from Linear 
          into a publishable forensic artifact. It uses the <strong>Ransom Note</strong> design system 
          to render systemic infiltration reports as editorial dossiers.
        </p>
      </div>

      <div className="now-staging-preview">
        <NowPageView meta={data.meta} sections={data.sections} />
      </div>
      
      <div className="now-staging-metadata">
        <div className="now-staging-metadata__inner">
          <span className="font-label text-[10px] font-bold uppercase tracking-widest text-secondary block mb-4">Linear Contract State</span>
          <pre className="text-[10px] bg-black/5 p-4 overflow-auto">
            {JSON.stringify(data.filterData, null, 2)}
          </pre>
        </div>
      </div>
    </div>
  );
};

<Loader />
